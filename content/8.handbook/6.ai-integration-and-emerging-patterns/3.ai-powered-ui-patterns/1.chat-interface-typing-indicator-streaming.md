# Chat Interface — Typing Indicator & Streaming

A polished AI chat interface is much more than a text box and a response. Users expect the responsiveness of consumer products like ChatGPT and Claude. This chapter covers building a production-quality chat UI with streaming output, typing indicators, message formatting, and smooth interactions.

## 1. Core Chat Component Architecture

```
ChatPage
  ├── MessageList           ← Scrollable message history
  │     ├── UserMessage     ← User messages
  │     └── AssistantMessage ← AI messages with streaming
  │           ├── MarkdownRenderer
  │           └── TypingIndicator (while streaming)
  ├── ChatInput             ← Input + send button + stop button
  └── ChatToolbar           ← New chat, model selector, etc.
```

## 2. Full Chat Implementation with `@ai-sdk/vue`

```vue
<!-- pages/chat.vue -->
<script setup lang="ts">
import { useChat } from '@ai-sdk/vue'

useHead({ title: 'AI Assistant' })

const { messages, input, handleSubmit, isLoading, error, reload, stop } = useChat({
  api: '/api/chat',
  initialMessages: [
    {
      id: 'welcome',
      role: 'assistant',
      content: 'Hello! I\'m your AI assistant. How can I help you today?'
    }
  ],
  onError: (err) => {
    toast.error('Something went wrong. Please try again.')
    console.error(err)
  }
})

// Auto-scroll to bottom when new messages arrive
const messageListRef = ref<HTMLElement | null>(null)
watch(messages, () => {
  nextTick(() => {
    messageListRef.value?.scrollTo({
      top: messageListRef.value.scrollHeight,
      behavior: 'smooth'
    })
  })
}, { deep: true })

// Submit on Enter (Shift+Enter for newline)
const onKeydown = (e: KeyboardEvent) => {
  if (e.key === 'Enter' && !e.shiftKey) {
    e.preventDefault()
    handleSubmit(e as any)
  }
}
</script>

<template>
  <div class="chat-page">
    <!-- Message list -->
    <div ref="messageListRef" class="message-list" role="log" aria-live="polite">
      <TransitionGroup name="message">
        <component
          :is="msg.role === 'user' ? UserMessage : AssistantMessage"
          v-for="msg in messages"
          :key="msg.id"
          :message="msg"
          :is-streaming="isLoading && msg === messages[messages.length - 1] && msg.role === 'assistant'"
        />
      </TransitionGroup>
    </div>

    <!-- Error state -->
    <div v-if="error" class="error-banner" role="alert">
      Failed to get response.
      <button @click="reload()">Retry</button>
    </div>

    <!-- Chat input -->
    <form class="chat-input-form" @submit="handleSubmit">
      <textarea
        v-model="input"
        :disabled="isLoading"
        placeholder="Message AI Assistant..."
        rows="1"
        @keydown="onKeydown"
      />
      <button v-if="isLoading" type="button" aria-label="Stop generating" @click="stop">
        <StopIcon />
      </button>
      <button v-else type="submit" :disabled="!input.trim()" aria-label="Send message">
        <SendIcon />
      </button>
    </form>
  </div>
</template>
```

## 3. Typing / Streaming Indicator

```vue
<!-- components/AssistantMessage.vue -->
<script setup lang="ts">
import { marked } from 'marked'
import DOMPurify from 'dompurify'

const props = defineProps<{
  message: { id: string; role: string; content: string }
  isStreaming: boolean
}>()

const renderedContent = computed(() =>
  DOMPurify.sanitize(marked.parse(props.message.content) as string)
)
</script>

<template>
  <div class="assistant-message" :class="{ streaming: isStreaming }">
    <div class="assistant-avatar">
      <AIIcon />
    </div>
    <div class="message-content">
      <!-- Render streamed content as markdown -->
      <div class="prose" v-html="renderedContent" />
      
      <!-- Animated cursor while streaming -->
      <span v-if="isStreaming" class="streaming-cursor" aria-hidden="true" />
      
      <!-- Typing indicator dots (before first token arrives) -->
      <div v-if="isStreaming && !message.content" class="typing-dots" aria-label="AI is thinking">
        <span /><span /><span />
      </div>
    </div>
  </div>
</template>

<style scoped>
.streaming-cursor {
  display: inline-block;
  width: 2px;
  height: 1em;
  background: currentColor;
  margin-left: 1px;
  animation: blink 1s step-end infinite;
}

}
</style>
```

## 5. Conversation Management

```typescript
// Save and restore conversations
const { messages, setMessages } = useChat({ api: '/api/chat' })

const saveConversation = () => {
  const saved = JSON.stringify(messages.value)
  localStorage.setItem(`chat-${conversationId}`, saved)
}

const loadConversation = (id: string) => {
  const saved = localStorage.getItem(`chat-${id}`)
  if (saved) setMessages(JSON.parse(saved))
}

// New conversation
const newConversation = () => {
  saveConversation() // Save current before clearing
  setMessages([])
}
```
