# Auto-completion Patterns

AI-powered auto-completion predicts and completes text as users type — from simple word suggestions to full paragraph completions. Done well, it dramatically speeds up user workflows. Done poorly, it interrupts and frustrates. This chapter covers the patterns for building effective, non-intrusive AI completion UIs.

## 1. Completion Types

| Type | Trigger | Scope | Example |
|---|---|---|---|
| **Word completion** | After 3+ chars | Next word | "Micro" → "Microsoft" |
| **Phrase completion** | After a clause | 5-20 words | "Send an email to" → "...all team members" |
| **Paragraph completion** | Pause in typing | Full paragraph | Entire next paragraph suggested |
| **Template completion** | Special trigger (`/`) | Structured template | `/meeting-notes` → full template |
| **Ghost text** | During typing | Rest of sentence | GitHub Copilot style |

## 2. Ghost Text Completion (Copilot Style)

```vue
<!-- components/GhostTextEditor.vue -->
<script setup lang="ts">
const content = ref('')
const ghost = ref('')
const isGhostLoading = ref(false)
const textarea = ref<HTMLTextAreaElement | null>(null)

const fetchCompletion = useDebounceFn(async () => {
  if (content.value.length < 10) { ghost.value = ''; return }
  
  isGhostLoading.value = true
  try {
    const { completion } = await $fetch<{ completion: string }>('/api/complete', {
      method: 'POST',
      body: {
        text: content.value,
        maxWords: 15  // Keep ghost text short
      }
    })
    ghost.value = completion
  } finally {
    isGhostLoading.value = false
  }
}, 800) // Wait 800ms after last keystroke

watch(content, (newVal, oldVal) => {
  // Clear ghost if user typed something different
  if (ghost.value && !newVal.endsWith(content.value.slice(-3))) ghost.value = ''
  fetchCompletion()
})

const acceptGhost = () => {
  if (ghost.value) {
    content.value += ghost.value
    ghost.value = ''
    nextTick(() => {
      textarea.value?.focus()
      textarea.value?.setSelectionRange(content.value.length, content.value.length)
    })
  }
}
</script>

<template>
  <div class="ghost-editor">
    <!-- Visual ghost text overlay -->
    <div class="editor-display" aria-hidden="true">
      <span class="editor-content">{{ content }}</span>
      <span class="ghost-text">{{ ghost }}</span>
    </div>
    
    <textarea
      ref="textarea"
      v-model="content"
      class="editor-input"
      placeholder="Start typing..."
      @keydown.tab.prevent="acceptGhost"
      @keydown.escape="ghost = ''"
    />
    
    <!-- Tab hint -->
    <div v-if="ghost && !isGhostLoading" class="tab-hint" aria-live="polite">
      Press <kbd>Tab</kbd> to accept suggestion
    </div>
  </div>
</template>

<style scoped>
.ghost-editor { position: relative; }

.editor-display {
  position: absolute; inset: 0;
  padding: /* match textarea padding */ 12px;
  pointer-events: none;
  white-space: pre-wrap;
  word-break: break-word;
}

.editor-content { visibility: hidden; } /* Invisible — just for sizing */
.ghost-text { color: hsl(215, 16%, 60%); font-style: italic; }
.editor-input { background: transparent; width: 100%; height: 100%; }
</style>
```

## 3. Server: Completion Endpoint

```typescript
// server/api/complete.post.ts
export default defineEventHandler(async (event) => {
  const { text, maxWords = 20 } = await readBody(event)

  const { text: completion } = await generateText({
    model: openai('gpt-4o-mini'),
    system: `You are a writing assistant. Complete the text naturally.
Output ONLY the completion (not the original text).
Keep completion to ${maxWords} words or less.
Match the writing style and tone of the input.`,
    prompt: text,
    maxTokens: maxWords * 2, // Rough tokens-to-words conversion
    temperature: 0.4,         // Lower = more predictable completions
    stopSequences: ['\n\n', '...', '?', '!'] // Stop at natural breakpoints
  })

  return { completion }
})
```

## 4. Slash Command Templates

```typescript
// composables/useSlashCommands.ts
const SLASH_COMMANDS = {
  '/meeting': () => `## Meeting Notes\nDate: ${new Date().toLocaleDateString()}\nAttendees: \n\n### Agenda\n1. \n\n### Action Items\n- `,
  '/email': () => `Subject: \n\nHi [Name],\n\n\n\nBest regards,\n${currentUser.name}`,
  '/bug-report': () => `## Bug Report\n\n**Description:**\n\n**Steps to Reproduce:**\n1. \n\n**Expected:** \n\n**Actual:** `,
}

export function useSlashCommands(content: Ref<string>) {
  const isSlashMenuOpen = ref(false)
  const slashQuery = ref('')

  watch(content, (val) => {
    const lastWord = val.split(/\s/).pop() ?? ''
    if (lastWord.startsWith('/') && lastWord.length > 1) {
      slashQuery.value = lastWord.slice(1)
      isSlashMenuOpen.value = true
    } else {
      isSlashMenuOpen.value = false
    }
  })

  const commands = computed(() =>
    Object.entries(SLASH_COMMANDS)
      .filter(([cmd]) => cmd.includes(slashQuery.value))
  )

  const applyCommand = (cmd: string) => {
    const template = SLASH_COMMANDS[cmd as keyof typeof SLASH_COMMANDS]?.()
    if (template) {
      // Replace the slash command with the template
      content.value = content.value.replace(new RegExp(`\\/${slashQuery.value}$`), template)
    }
    isSlashMenuOpen.value = false
  }

  return { isSlashMenuOpen, commands, applyCommand }
}
```
