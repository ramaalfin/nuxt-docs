# Suggestion System — Inference at Edge

A suggestion system provides real-time AI-generated recommendations as the user types or interacts with your application. Running inference at the edge (Cloudflare Workers, Vercel Edge Functions) is critical for achieving the sub-200ms latency that makes suggestions feel responsive rather than intrusive.

## 1. Edge Inference Architecture

```
User types in input
  ↓ (300ms debounce)
Edge Function (Cloudflare/Vercel)
  ↓ (fast — close to user)
Small/Fast LLM (gpt-4o-mini, llama-3.1-8b)
  ↓ (~100-200ms)
3-5 suggestions returned
  ↓
UI renders suggestions inline
```

**Key rule:** Use small, fast models for suggestions. A 2-second wait kills the UX. `gpt-4o-mini` with streaming typically produces the first token in < 150ms from the edge.

## 2. Edge API Route

```typescript
// server/api/suggestions.post.ts
// Deploy this to edge runtime for minimum latency
export default defineEventHandler(async (event) => {
  const { input, context, type = 'text' } = await readBody(event)
  
  if (!input || input.trim().length < 3) return { suggestions: [] }

  const systemPrompts = {
    text: 'Generate 3 short text completions. Return JSON array of strings only.',
    search: 'Suggest 5 related search queries. Return JSON array of strings only.',
    code: 'Complete the code. Return only the completion, no explanation.',
    tags: 'Suggest 5 relevant tags. Return JSON array of short strings.'
  }

  const { text } = await generateText({
    model: openai('gpt-4o-mini'),
    system: systemPrompts[type as keyof typeof systemPrompts] ?? systemPrompts.text,
    prompt: `Input: "${input}"\nContext: ${context ?? 'none'}`,
    maxTokens: 150,
    temperature: 0.3 // Lower temperature for consistent, predictable suggestions
  })

  try {
    return { suggestions: JSON.parse(text) }
  } catch {
    // Fallback if model didn't return valid JSON
    return { suggestions: text.split('\n').filter(s => s.trim()).slice(0, 5) }
  }
})
```

## 3. Client-Side Suggestion Composable

```typescript
// composables/useSuggestions.ts
export function useSuggestions(type: 'text' | 'search' | 'tags' = 'text') {
  const input = ref('')
  const suggestions = ref<string[]>([])
  const isLoading = ref(false)
  const selectedIndex = ref(-1)

  const fetchSuggestions = useDebounceFn(async (value: string) => {
    if (value.trim().length < 3) {
      suggestions.value = []
      return
    }

    isLoading.value = true
    try {
      const { suggestions: data } = await $fetch<{ suggestions: string[] }>('/api/suggestions', {
        method: 'POST',
        body: { input: value, type }
      })
      suggestions.value = data
    } finally {
      isLoading.value = false
    }
  }, 300)

  watch(input, fetchSuggestions)

  const selectSuggestion = (suggestion: string) => {
    input.value = suggestion
    suggestions.value = []
    selectedIndex.value = -1
  }

  // Keyboard navigation
  const onKeydown = (e: KeyboardEvent) => {
    if (!suggestions.value.length) return
    if (e.key === 'ArrowDown') {
      e.preventDefault()
      selectedIndex.value = Math.min(selectedIndex.value + 1, suggestions.value.length - 1)
    }
    if (e.key === 'ArrowUp') {
      e.preventDefault()
      selectedIndex.value = Math.max(selectedIndex.value - 1, -1)
    }
    if (e.key === 'Enter' && selectedIndex.value >= 0) {
      e.preventDefault()
      selectSuggestion(suggestions.value[selectedIndex.value])
    }
    if (e.key === 'Escape') suggestions.value = []
  }

  return { input, suggestions, isLoading, selectedIndex, selectSuggestion, onKeydown }
}
```

## 4. Suggestion UI Component

```vue
<!-- components/SmartInput.vue -->
<script setup lang="ts">
const { input, suggestions, isLoading, selectedIndex, selectSuggestion, onKeydown } = useSuggestions('search')
</script>

<template>
  <div class="smart-input" role="combobox" :aria-expanded="suggestions.length > 0">
    <input
      v-model="input"
      type="text"
      placeholder="Search..."
      aria-autocomplete="list"
      :aria-activedescendant="selectedIndex >= 0 ? `suggestion-${selectedIndex}` : undefined"
      @keydown="onKeydown"
    />
    
    <SpinnerIcon v-if="isLoading" class="input-spinner" aria-hidden="true" />

    <!-- Suggestions dropdown -->
    <Transition name="fade">
      <ul v-if="suggestions.length > 0" class="suggestions-list" role="listbox">
        <li
          v-for="(suggestion, i) in suggestions"
          :id="`suggestion-${i}`"
          :key="suggestion"
          role="option"
          :aria-selected="selectedIndex === i"
          :class="{ active: selectedIndex === i }"
          @click="selectSuggestion(suggestion)"
          @mouseenter="selectedIndex = i"
        >
          <SparklesIcon class="suggestion-icon" aria-hidden="true" />
          {{ suggestion }}
        </li>
      </ul>
    </Transition>
  </div>
</template>
```

## 5. Inline Ghost Text (GitHub Copilot Style)

```vue
<!-- Ghost text completion shown inline in the input -->
<script setup lang="ts">
const text = ref('')
const ghost = ref('')

// Fetch ghost completion on pause
const fetchGhost = useDebounceFn(async () => {
  if (text.value.length < 5) { ghost.value = ''; return }
  const { completion } = await $fetch('/api/ghost-complete', {
    method: 'POST', body: { text: text.value }
  })
  ghost.value = completion
}, 600)

watch(text, fetchGhost)

const acceptGhost = () => {
  if (ghost.value) { text.value += ghost.value; ghost.value = '' }
}
</script>

<template>
  <div class="ghost-input">
    <div class="ghost-display">
      <span class="text-visible">{{ text }}</span>
      <span class="text-ghost" aria-hidden="true">{{ ghost }}</span>
      <!-- Tab to accept ghost text -->
      <span v-if="ghost" class="tab-hint">Tab</span>
    </div>
    <textarea v-model="text" @keydown.tab.prevent="acceptGhost" />
  </div>
</template>
```
