# Progressive Synchronization

Progressive synchronization is the strategy of syncing data in stages based on priority and user context — rather than syncing everything at once when connectivity is restored. This improves perceived performance, reduces bandwidth usage, and ensures users see the most important data first.

## 1. Sync Priority Tiers

Define clear priority tiers for your data:

```typescript
// types/sync.ts
export enum SyncPriority {
  CRITICAL = 1,   // Must sync immediately (user mutations, financial data)
  HIGH = 2,       // Sync as soon as possible (recent activity, current page data)
  NORMAL = 3,     // Sync in background (historical data, settings)
  LOW = 4         // Sync opportunistically (analytics, prefetch)
}
```

## 2. Tiered Sync Manager

```typescript
// composables/useProgressiveSyncManager.ts
interface SyncTask {
  id: string
  priority: SyncPriority
  syncFn: () => Promise<void>
  lastSyncedAt?: number
  intervalMs?: number // For periodic re-sync
}

export function useProgressiveSyncManager() {
  const isSyncing = ref(false)
  const currentTier = ref<SyncPriority | null>(null)
  const tasks: SyncTask[] = []

  const register = (task: SyncTask) => tasks.push(task)

  const runSync = async () => {
    if (isSyncing.value || !navigator.onLine) return
    isSyncing.value = true

    // Process tiers in order — critical first, low last
    for (const priority of [
      SyncPriority.CRITICAL,
      SyncPriority.HIGH,
      SyncPriority.NORMAL,
      SyncPriority.LOW
    ]) {
      currentTier.value = priority

      const tierTasks = tasks
        .filter(t => t.priority === priority)
        .filter(t => {
          // Only run if interval has passed (or never run before)
          if (!t.intervalMs || !t.lastSyncedAt) return true
          return Date.now() - t.lastSyncedAt > t.intervalMs
        })

      // Run tasks within a tier in parallel
      await Promise.allSettled(
        tierTasks.map(async task => {
          try {
            await task.syncFn()
            task.lastSyncedAt = Date.now()
          } catch (error) {
            console.error(`Sync task ${task.id} failed:`, error)
          }
        })
      )

      // Yield to the browser between tiers to prevent blocking UI
      await new Promise(resolve => setTimeout(resolve, 0))
    }

    currentTier.value = null
    isSyncing.value = false
  }

  // Trigger sync on reconnect
  useEventListener(window, 'online', runSync)

  return { register, runSync, isSyncing, currentTier }
}
```

## 3. Example: Registering Sync Tasks

```typescript
// plugins/sync.client.ts
export default defineNuxtPlugin(() => {
  const { register, runSync } = useProgressiveSyncManager()
  const { processBatch: processMutationQueue } = useSyncWorker()
  const queryClient = useQueryClient()

  // CRITICAL: User's own queued mutations (must sync first)
  register({
    id: 'mutation-queue',
    priority: SyncPriority.CRITICAL,
    syncFn: processMutationQueue
  })

  // HIGH: Current page data
  register({
    id: 'active-document',
    priority: SyncPriority.HIGH,
    syncFn: async () => {
      const activeDocId = useActiveDocStore().currentDocId
      if (activeDocId) {
        await queryClient.refetchQueries({ queryKey: ['document', activeDocId] })
      }
    }
  })

  // NORMAL: User's recent activity and notifications
  register({
    id: 'notifications',
    priority: SyncPriority.NORMAL,
    intervalMs: 1000 * 60 * 5, // Re-sync every 5 minutes
    syncFn: async () => {
      await queryClient.refetchQueries({ queryKey: ['notifications'] })
    }
  })

  // LOW: Prefetch next likely page (background, opportunistic)
  register({
    id: 'prefetch-next',
    priority: SyncPriority.LOW,
    intervalMs: 1000 * 60 * 15,
    syncFn: async () => {
      // Prefetch data for the most likely next navigation
      await queryClient.prefetchQuery({ queryKey: ['dashboard'], queryFn: fetchDashboard })
    }
  })

  // Start initial sync
  if (navigator.onLine) runSync()
})
```

## 4. Sync Progress UI

Show users what tier is currently syncing so they understand what's happening.

```vue
<template>
  <div v-if="isSyncing" class="sync-progress">
    <div class="sync-bar">
      <div
        class="sync-progress-fill"
        :style="{ width: `${syncProgress}%` }"
      />
    </div>
    <span class="sync-label">
      <template v-if="currentTier === SyncPriority.CRITICAL">Syncing your changes...</template>
      <template v-else-if="currentTier === SyncPriority.HIGH">Refreshing current data...</template>
      <template v-else-if="currentTier === SyncPriority.NORMAL">Updating notifications...</template>
      <template v-else>Completing sync...</template>
    </span>
  </div>
</template>
```

## 5. Background Sync with Service Worker

For the most resilient sync experience, trigger sync from the Service Worker even when the app tab is not open.

```typescript
// sw.ts (Service Worker)
self.addEventListener('sync', (event: SyncEvent) => {
  if (event.tag === 'mutation-queue') {
    event.waitUntil(
      // Send pending operations from IndexedDB to the server
      processMutationQueueInSW()
    )
  }
})
```

```typescript
// Register background sync when mutations are queued
const registerBackgroundSync = async () => {
  const registration = await navigator.serviceWorker.ready
  if ('sync' in registration) {
    await (registration as any).sync.register('mutation-queue')
  }
}
```
