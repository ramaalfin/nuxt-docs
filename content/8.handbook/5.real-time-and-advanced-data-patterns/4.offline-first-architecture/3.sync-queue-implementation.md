# Sync Queue Implementation

A sync queue is the mechanism that captures local changes made while offline and reliably delivers them to the server when connectivity is restored. It is the bridge between offline-first local state and the server's canonical state.

## 1. The Sync Queue Architecture

```
User Action (offline)
  ↓
1. Apply to local IndexedDB (instant UI feedback)
2. Add to Sync Queue (persisted in IndexedDB)

Network restored
  ↓
3. Sync Queue Worker runs
4. Dequeues operations in order
5. Sends to server
6. On success: marks as synced, removes from queue
7. On failure: retries with exponential backoff
8. On conflict: marks as conflict, notifies user
```

## 2. Queue Schema (Dexie)

```typescript
// db/index.ts
interface SyncQueueItem {
  id?: number           // Auto-incremented
  operationId: string   // UUID for deduplication
  endpoint: string      // '/api/tasks/123'
  method: 'POST' | 'PUT' | 'PATCH' | 'DELETE'
  body?: any
  entityType: string    // 'task', 'comment', 'order' — for conflict resolution
  entityId: string
  retries: number
  createdAt: number
  status: 'pending' | 'in-flight' | 'failed' | 'conflict'
  errorMessage?: string
}

class AppDatabase extends Dexie {
  syncQueue!: EntityTable<SyncQueueItem, 'id'>
  
  constructor() {
    super('app-db')
    this.version(1).stores({
      tasks: 'id, status, updatedAt',
      syncQueue: '++id, status, entityType, entityId, createdAt'
    })
  }
}

export const db = new AppDatabase()
```

## 3. Queue Operations

```typescript
// composables/useSyncQueue.ts
export function useSyncQueue() {
  
  const enqueue = async (operation: Omit<SyncQueueItem, 'id' | 'retries' | 'createdAt' | 'status'>) => {
    await db.syncQueue.add({
      ...operation,
      retries: 0,
      createdAt: Date.now(),
      status: 'pending'
    })
  }

  const getPendingOps = () => {
    return db.syncQueue
      .where('status').anyOf(['pending', 'failed'])
      .and(op => op.retries < 5) // Max 5 retries
      .sortBy('createdAt') // Process in order
  }

  const markSynced = (id: number) => db.syncQueue.delete(id)
  
  const markFailed = (id: number, error: string) =>
    db.syncQueue.update(id, {
      status: 'failed',
      errorMessage: error,
      retries: db.syncQueue.get(id).then(op => (op?.retries ?? 0) + 1) as any
    })

  const getConflicts = () => db.syncQueue.where('status').equals('conflict').toArray()
  
  return { enqueue, getPendingOps, markSynced, markFailed, getConflicts }
}
```

## 4. Sync Worker

```typescript
// composables/useSyncWorker.ts
export function useSyncWorker() {
  const { enqueue, getPendingOps, markSynced, markFailed } = useSyncQueue()
  const isSyncing = ref(false)
  let syncTimer: ReturnType<typeof setTimeout> | null = null

  const processBatch = async () => {
    if (isSyncing.value || !navigator.onLine) return
    
    const pending = await getPendingOps()
    if (pending.length === 0) return

    isSyncing.value = true
    
    for (const op of pending) {
      try {
        await db.syncQueue.update(op.id!, { status: 'in-flight' })
        
        await $fetch(op.endpoint, {
          method: op.method,
          body: op.body,
          // Send the entity's last known version for conflict detection
          headers: { 'X-Entity-Version': op.entityId }
        })
        
        await markSynced(op.id!)
      } catch (error: any) {
        if (error.status === 409) {
          // Conflict — don't retry, flag for user resolution
          await db.syncQueue.update(op.id!, { status: 'conflict' })
        } else {
          // Other error — mark for retry
          await db.syncQueue.update(op.id!, {
            status: 'failed',
            retries: (op.retries ?? 0) + 1,
            errorMessage: error.message
          })
        }
      }
    }
    
    isSyncing.value = false
  }

  // Sync when network comes back online
  useEventListener(window, 'online', () => {
    processBatch()
  })

  // Also sync periodically while online
  const startPeriodicSync = () => {
    syncTimer = setInterval(processBatch, 30_000) // Every 30 seconds
  }

  onMounted(() => {
    startPeriodicSync()
    if (navigator.onLine) processBatch() // Sync immediately on mount
  })
  
  onUnmounted(() => syncTimer && clearInterval(syncTimer))

  return { isSyncing, processBatch }
}
```

## 5. Mutation Wrapper with Queue Fallback

```typescript
// composables/useOfflineMutation.ts
export function useOfflineMutation<T>(endpoint: string, method: 'POST' | 'PATCH' | 'DELETE') {
  const { enqueue } = useSyncQueue()

  const mutate = async (body: T, entityId: string, entityType: string) => {
    if (navigator.onLine) {
      // Online: send directly
      return $fetch(endpoint, { method, body: body as any })
    } else {
      // Offline: apply locally and queue for later
      await enqueue({
        operationId: crypto.randomUUID(),
        endpoint,
        method,
        body,
        entityType,
        entityId
      })
    }
  }

  return { mutate }
}
```
