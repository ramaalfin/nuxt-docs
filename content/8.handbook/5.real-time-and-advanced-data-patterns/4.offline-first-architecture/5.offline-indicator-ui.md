# Offline Indicator UI

An offline indicator communicates to users that their network connection has been lost and that the application is running in offline mode. A well-designed offline indicator is subtle enough not to alarm users unnecessarily, but clear enough to explain why some features might not work.

## 1. Detecting Online/Offline Status

```typescript
// composables/useNetworkStatus.ts
export function useNetworkStatus() {
  const isOnline = ref(navigator.onLine)
  const isSlowConnection = ref(false)
  const connectionType = ref<string>('unknown')

  // Listen for browser online/offline events
  useEventListener(window, 'online', () => { isOnline.value = true })
  useEventListener(window, 'offline', () => { isOnline.value = false })

  // Detect connection quality (if supported)
  const connection = (navigator as any).connection
  if (connection) {
    connectionType.value = connection.effectiveType // '4g', '3g', '2g', 'slow-2g'
    isSlowConnection.value = ['slow-2g', '2g'].includes(connection.effectiveType)

    connection.addEventListener('change', () => {
      connectionType.value = connection.effectiveType
      isSlowConnection.value = ['slow-2g', '2g'].includes(connection.effectiveType)
    })
  }

  // Note: navigator.onLine is not 100% reliable (it can say "online" on a captive portal)
  // Perform a lightweight health check for a more reliable signal
  const verifyConnectivity = async () => {
    try {
      await $fetch('/api/health', { timeout: 5000 })
      isOnline.value = true
    } catch {
      isOnline.value = false
    }
  }

  // Verify every 30 seconds
  const interval = setInterval(verifyConnectivity, 30_000)
  onUnmounted(() => clearInterval(interval))

  return { isOnline, isSlowConnection, connectionType }
}
```

## 2. The Offline Indicator Component

```vue
<!-- components/OfflineIndicator.vue -->
<script setup lang="ts">
const { isOnline, isSlowConnection } = useNetworkStatus()
const { pendingCount } = useSyncQueue()

// Show "back online" message briefly when reconnecting
const showReconnected = ref(false)

watch(isOnline, (online, wasOnline) => {
  if (online && !wasOnline) {
    showReconnected.value = true
    setTimeout(() => { showReconnected.value = false }, 3000)
  }
})
</script>

<template>
  <!-- Offline Banner -->
  <Transition name="slide-down">
    <div
      v-if="!isOnline"
      class="offline-banner"
      role="status"
      aria-live="polite"
    >
      <WifiOffIcon aria-hidden="true" />
      <div>
        <strong>You're offline</strong>
        <span v-if="pendingCount > 0">
          {{ pendingCount }} change{{ pendingCount > 1 ? 's' : '' }} will sync when you reconnect.
        </span>
        <span v-else>Changes you make will be saved locally.</span>
      </div>
    </div>
  </Transition>

  <!-- Back Online Toast -->
  <Transition name="slide-down">
    <div
      v-if="showReconnected"
      class="reconnected-toast"
      role="status"
    >
      <CheckCircleIcon aria-hidden="true" />
      Back online! Syncing your changes...
    </div>
  </Transition>

  <!-- Slow Connection Warning (subtle) -->
  <div v-if="isOnline && isSlowConnection" class="slow-connection-badge">
    <SignalLowIcon />
    Slow connection
  </div>
</template>

<style scoped>

</style>
```

## 3. Feature Availability Indicators

Show which features are unavailable offline instead of silently failing.

```vue
<!-- Disable features that require connectivity -->
<template>
  <div>
    <button
      :disabled="!isOnline"
      :title="!isOnline ? 'Available when online' : undefined"
      @click="shareDocument"
    >
      <ShareIcon />
      Share
      <span v-if="!isOnline" class="text-xs opacity-60">(offline)</span>
    </button>

    <!-- Read-only operations still work offline -->
    <button @click="viewDocument">
      <EyeIcon />
      View
    </button>
  </div>
</template>
```

## 4. Sync Status Indicator

Show the number of pending sync operations in the UI so users know their changes will be preserved.

```vue
<template>
  <div class="sync-status" :class="syncStatusClass">
    <template v-if="isSyncing">
      <ArrowPathIcon class="animate-spin" />
      Syncing...
    </template>
    <template v-else-if="pendingCount > 0">
      <CloudArrowUpIcon />
      {{ pendingCount }} pending
    </template>
    <template v-else-if="conflictCount > 0">
      <ExclamationTriangleIcon />
      {{ conflictCount }} conflict{{ conflictCount > 1 ? 's' : '' }}
    </template>
    <template v-else>
      <CheckCircleIcon />
      All synced
    </template>
  </div>
</template>
```
