# TanStack Query v5 Mastery (Vue Query)

TanStack Query (formerly React Query) is the gold standard for server state management. Its Vue adapter, **Vue Query** (`@tanstack/vue-query`), brings the full power of TanStack Query v5 to Nuxt applications—eliminating the need for Pinia to manage server-fetched data.

## 1. Installation & Setup in Nuxt

```bash
pnpm add @tanstack/vue-query
```

```typescript
// plugins/vue-query.ts
import { VueQueryPlugin, QueryClient, hydrate, dehydrate } from '@tanstack/vue-query'

export default defineNuxtPlugin((nuxtApp) => {
  const queryClient = new QueryClient({
    defaultOptions: {
      queries: {
        staleTime: 1000 * 60 * 5, // 5 minutes — data is "fresh" for 5 mins
        gcTime: 1000 * 60 * 10,   // 10 minutes — cache kept for 10 mins after last use
        retry: 2,
        refetchOnWindowFocus: true,
      }
    }
  })

  // SSR: Dehydrate query cache and send to client
  nuxtApp.hook('app:rendered', () => {
    nuxtApp.payload.vueQueryState = dehydrate(queryClient)
  })

  // Client: Rehydrate from server-sent state (avoids refetch on hydration)
  nuxtApp.hook('app:created', () => {
    hydrate(queryClient, nuxtApp.payload.vueQueryState)
  })

  nuxtApp.vueApp.use(VueQueryPlugin, { queryClient })
})
```

## 2. Basic Query Usage

```typescript
// composables/useProducts.ts
import { useQuery } from '@tanstack/vue-query'

export function useProducts(categoryId: Ref<string>) {
  return useQuery({
    // Query key — uniquely identifies this query in the cache
    // Changing categoryId.value automatically triggers a refetch
    queryKey: ['products', categoryId],
    
    queryFn: () => $fetch(`/api/products?category=${categoryId.value}`),
    
    // Options
    enabled: computed(() => !!categoryId.value), // Only fetch if categoryId is set
    staleTime: 1000 * 60 * 2, // Override global default — 2 minute freshness
  })
}
```

```vue
<script setup lang="ts">
const categoryId = ref('electronics')
const { data: products, isPending, isError, error, refetch } = useProducts(categoryId)
</script>

<template>
  <div v-if="isPending"><ProductSkeleton /></div>
  <div v-else-if="isError">
    Error: {{ error.message }}
    <button @click="refetch()">Retry</button>
  </div>
  <ProductGrid v-else :products="data" />
</template>
```

## 3. Prefetching for Instant Navigation

```typescript
// Prefetch on hover to make navigation feel instant
const queryClient = useQueryClient()

const prefetchProduct = (productId: string) => {
  queryClient.prefetchQuery({
    queryKey: ['product', productId],
    queryFn: () => $fetch(`/api/products/${productId}`),
    staleTime: 1000 * 60 * 5,
  })
}
```

```vue
<NuxtLink
  :to="`/products/${product.id}`"
  @mouseenter="prefetchProduct(product.id)"
>
  {{ product.name }}
</NuxtLink>
```

## 4. useInfiniteQuery for Pagination

```typescript
// composables/useInfiniteProducts.ts
import { useInfiniteQuery } from '@tanstack/vue-query'

export function useInfiniteProducts(search: Ref<string>) {
  return useInfiniteQuery({
    queryKey: ['products', 'infinite', search],
    queryFn: ({ pageParam }) => $fetch('/api/products', {
      query: { page: pageParam, search: search.value, limit: 20 }
    }),
    initialPageParam: 1,
    getNextPageParam: (lastPage, allPages) => {
      return lastPage.hasNextPage ? allPages.length + 1 : undefined
    }
  })
}
```

```vue
<script setup lang="ts">
const { data, fetchNextPage, hasNextPage, isFetchingNextPage } = useInfiniteProducts(search)

const allProducts = computed(() => data.value?.pages.flatMap(p => p.items) ?? [])

// Trigger on scroll to bottom
const { stop } = useIntersectionObserver(loadMoreTrigger, ([entry]) => {
  if (entry.isIntersecting && hasNextPage.value && !isFetchingNextPage.value) {
    fetchNextPage()
  }
})
</script>
```

## 5. Global Query Key Management

Use query key factories to keep cache keys consistent and type-safe.

```typescript
// queries/keys.ts
export const queryKeys = {
  products: {
    all: ['products'] as const,
    lists: () => [...queryKeys.products.all, 'list'] as const,
    list: (filters: ProductFilters) => [...queryKeys.products.lists(), filters] as const,
    detail: (id: string) => [...queryKeys.products.all, 'detail', id] as const,
  },
  user: {
    profile: (id: string) => ['user', id, 'profile'] as const,
    orders: (id: string) => ['user', id, 'orders'] as const,
  }
}
```

```typescript
// Invalidate all product list queries after a mutation
queryClient.invalidateQueries({ queryKey: queryKeys.products.lists() })
```
