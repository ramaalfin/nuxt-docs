# Optimistic Updates & Rollback

Optimistic updates immediately apply a mutation to the local UI state before the server responds, giving the user instant feedback. If the server request succeeds, the optimistic update is confirmed. If it fails, the state is rolled back to its previous value and the user is informed.

When implemented well, optimistic updates make an app feel instantaneous regardless of network conditions.

## 1. The Pattern

```
1. User clicks "Like"
2. Immediately: likeCount++ + isLiked = true (optimistic)
3. Send POST /api/posts/123/like to server
4a. Server succeeds → optimistic state confirmed ✅
4b. Server fails → rollback to original state + show error ❌
```

## 2. TanStack Query Optimistic Updates

TanStack Query's `useMutation` provides first-class support for optimistic updates via `onMutate`, `onError`, and `onSettled` lifecycle hooks.

```typescript
// composables/useOptimisticLike.ts
export function useOptimisticLike(postId: Ref<string>) {
  const queryClient = useQueryClient()

  return useMutation({
    mutationFn: () => $fetch(`/api/posts/${postId.value}/like`, { method: 'POST' }),

    // 1. Called BEFORE the mutation fires
    onMutate: async () => {
      // Cancel any in-flight refetches (they would overwrite our optimistic update)
      await queryClient.cancelQueries({ queryKey: ['post', postId.value] })

      // Snapshot the current value (for rollback)
      const previousPost = queryClient.getQueryData<Post>(['post', postId.value])

      // Optimistically update the cache
      queryClient.setQueryData<Post>(['post', postId.value], (old) => {
        if (!old) return old
        return {
          ...old,
          isLiked: !old.isLiked,
          likeCount: old.isLiked ? old.likeCount - 1 : old.likeCount + 1
        }
      })

      // Return context containing the snapshot (available in onError)
      return { previousPost }
    },

    // 2. Called if the mutation fails — roll back to the snapshot
    onError: (err, variables, context) => {
      queryClient.setQueryData(['post', postId.value], context?.previousPost)
      toast.error('Failed to update like. Please try again.')
    },

    // 3. Called when the mutation is settled (success or error)
    // Always sync with server truth to correct any discrepancy
    onSettled: () => {
      queryClient.invalidateQueries({ queryKey: ['post', postId.value] })
    }
  })
}
```

## 3. List Optimistic Updates (Add/Remove)

```typescript
// Optimistic task creation in a list
const { mutate: createTask } = useMutation({
  mutationFn: (newTask: CreateTaskInput) =>
    $fetch('/api/tasks', { method: 'POST', body: newTask }),

  onMutate: async (newTask) => {
    await queryClient.cancelQueries({ queryKey: ['tasks'] })
    const previousTasks = queryClient.getQueryData<Task[]>(['tasks'])

    // Optimistically add the new task with a temporary ID
    queryClient.setQueryData<Task[]>(['tasks'], (old) => [
      ...(old ?? []),
      {
        ...newTask,
        id: `temp-${Date.now()}`, // Will be replaced by server's real ID
        isOptimistic: true         // Flag for UI (show loading indicator on this item)
      }
    ])

    return { previousTasks }
  },

  onError: (err, newTask, context) => {
    queryClient.setQueryData(['tasks'], context?.previousTasks)
    toast.error('Failed to create task')
  },

  onSuccess: (serverTask) => {
    // Replace the temp item with the real server-returned item
    queryClient.setQueryData<Task[]>(['tasks'], (old) =>
      old?.map(t => t.id.startsWith('temp-') ? serverTask : t)
    )
  }
})
```

## 4. Optimistic Updates Without TanStack Query (Pinia)

```typescript
// stores/cart.ts
export const useCartStore = defineStore('cart', () => {
  const items = ref<CartItem[]>([])

  const removeItem = async (itemId: string) => {
    // Save previous state
    const previousItems = [...items.value]

    // Optimistically remove
    items.value = items.value.filter(i => i.id !== itemId)

    try {
      await $fetch(`/api/cart/${itemId}`, { method: 'DELETE' })
    } catch {
      // Rollback
      items.value = previousItems
      toast.error('Could not remove item. Please try again.')
    }
  }

  return { items, removeItem }
})
```

## 5. UI Patterns for Optimistic State

```vue
<template>
  <li
    v-for="task in tasks"
    :key="task.id"
    :class="{ 'opacity-60 pointer-events-none': task.isOptimistic }"
  >
    {{ task.title }}
    <!-- Show spinner on optimistic items still waiting for server confirmation -->
    <MiniSpinner v-if="task.isOptimistic" />
  </li>
</template>
```
