# Backpressure & Message Queuing

Backpressure is what happens when a message producer sends data faster than the consumer can process it. In real-time systems, uncontrolled message flooding can crash browsers (by exhausting memory), destabilize servers, and create a cascading degradation that takes down your entire system.

## 1. Understanding Backpressure in WebSockets

The WebSocket `send()` method is non-blocking — it enqueues messages into the browser's internal send buffer. If you call `send()` faster than the network can drain the buffer, `socket.bufferedAmount` grows until memory is exhausted.

```typescript
// Check if the WebSocket send buffer is backing up
const send = (data: object) => {
  if (!socket.value || socket.value.readyState !== WebSocket.OPEN) return
  
  const MAX_BUFFERED_AMOUNT = 64 * 1024 // 64KB threshold
  
  if (socket.value.bufferedAmount > MAX_BUFFERED_AMOUNT) {
    console.warn('[WS] Send buffer full, dropping message to prevent backpressure')
    // For critical messages, queue them; for ephemeral ones (cursor), drop them
    return
  }
  
  socket.value.send(JSON.stringify(data))
}
```

## 2. Client-Side Message Queue with Priority

Not all messages are equal. Critical messages (chat, order submissions) should never be dropped, while ephemeral messages (cursor position, heartbeat) can be dropped without consequence.

```typescript
// composables/usePriorityMessageQueue.ts
type Priority = 'critical' | 'normal' | 'ephemeral'

interface QueuedMessage {
  data: object
  priority: Priority
  enqueuedAt: number
}

export function usePriorityMessageQueue(socket: Ref<WebSocket | null>) {
  const queues: Record<Priority, QueuedMessage[]> = {
    critical: [],
    normal: [],
    ephemeral: []
  }

  const enqueue = (data: object, priority: Priority = 'normal') => {
    queues[priority].push({ data, priority, enqueuedAt: Date.now() })
  }

  // Flush queue in priority order on each animation frame
  const flush = () => {
    if (socket.value?.readyState !== WebSocket.OPEN) {
      requestAnimationFrame(flush)
      return
    }

    const MAX_BUFFER = 32 * 1024

    // Drain critical queue first, then normal, then ephemeral
    for (const priority of ['critical', 'normal', 'ephemeral'] as Priority[]) {
      while (queues[priority].length > 0) {
        if (socket.value.bufferedAmount > MAX_BUFFER) break // Apply backpressure

        const message = queues[priority].shift()!

        // Drop stale ephemeral messages (older than 100ms)
        if (priority === 'ephemeral' && Date.now() - message.enqueuedAt > 100) continue

        socket.value.send(JSON.stringify(message.data))
      }
    }

    requestAnimationFrame(flush)
  }

  onMounted(() => requestAnimationFrame(flush))

  return { enqueue }
}
```

## 3. Server-Side Rate Limiting & Backpressure

Protect the server from clients that send messages too rapidly (malicious or buggy clients).

```typescript
// server/routes/ws.ts
const clientRateLimits = new Map<string, { count: number; resetAt: number }>()

const checkRateLimit = (peerId: string, maxPerSecond = 50): boolean => {
  const now = Date.now()
  const limit = clientRateLimits.get(peerId)

  if (!limit || now > limit.resetAt) {
    clientRateLimits.set(peerId, { count: 1, resetAt: now + 1000 })
    return true
  }

  if (limit.count >= maxPerSecond) {
    return false // Rate limit exceeded
  }

  limit.count++
  return true
}

export default defineWebSocketHandler({
  message(peer, message) {
    if (!checkRateLimit(peer.id)) {
      peer.send(JSON.stringify({
        type: 'ERROR',
        code: 'RATE_LIMITED',
        message: 'You are sending messages too fast. Please slow down.'
      }))
      return // Drop the message
    }

    // Process message normally
    handleMessage(peer, JSON.parse(message.text()))
  }
})
```

## 4. Message Deduplication

In unstable networks, the same message can be delivered multiple times (especially with at-least-once delivery guarantees). Use sequence numbers or UUIDs to deduplicate.

```typescript
// composables/useDeduplicatedMessages.ts
export function useDeduplicatedMessages() {
  const processedIds = new Set<string>()
  const MAX_CACHE_SIZE = 1000

  const processMessage = (message: { id: string; [key: string]: any }) => {
    if (processedIds.has(message.id)) {
      return false // Duplicate, skip processing
    }
    
    processedIds.add(message.id)
    
    // Prevent the set from growing unboundedly
    if (processedIds.size > MAX_CACHE_SIZE) {
      const firstEntry = processedIds.values().next().value
      processedIds.delete(firstEntry)
    }
    
    return true // New message, process it
  }

  return { processMessage }
}
```

## 5. Dead Letter Queue for Failed Messages

For critical messages that fail to send (e.g., during reconnection), queue them locally and retry.

```typescript
export function useDeadLetterQueue(send: (data: object) => boolean) {
  const deadLetterQueue = useLocalStorage<Array<{ data: object; attempts: number }>>('dlq', [])
  
  const sendWithFallback = (data: object) => {
    const success = send(data)
    if (!success) {
      deadLetterQueue.value.push({ data, attempts: 0 })
    }
  }

  // Retry dead letters when connection is restored
  const replayDeadLetters = async () => {
    const pending = deadLetterQueue.value.splice(0)
    for (const item of pending) {
      if (!send(item.data)) {
        item.attempts++
        if (item.attempts < 5) {
          deadLetterQueue.value.push(item) // Re-enqueue if still failing
        }
      }
    }
  }

  return { sendWithFallback, replayDeadLetters }
}
```
